def cleanup
  begin
    self.interacting = false if self.interactive?
    rstream.close if (rstream)
  rescue ::Exception
  end

  rstream = nil
  super
end

def initialize(rstream, opts={})
  # A nil is passed in the case of non-stream interactive sessions (Meterpreter)
  if rstream
    self.rstream = rstream
    klass = opts[:udp_session] ? Rex::IO::RingBufferUdp : Rex::IO::RingBuffer
    self.ring    = klass.new(rstream, {:size => opts[:ring_size] || 100 })
  end
  super()
end

def tunnel_local
  return @local_info if @local_info
  begin
    @local_info = rstream.localinfo
  rescue ::Exception
    @local_info = '127.0.0.1'
  end
end
