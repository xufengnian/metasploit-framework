def load_rdi_dll(dll_path)
  dll = ''
  ::File.open(dll_path, 'rb') { |f| dll = f.read }

  offset = parse_pe(dll)

  unless offset
    raise "Cannot find the ReflectiveLoader entry point in #{dll_path}"
  end

  return dll, offset
end
